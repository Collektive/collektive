const highlightMatch = (text, searchTerm) => {
    if (!searchTerm) return text; 
    const escapedSearchTerm = searchTerm.replace(/[-/\\^$*+?.()|[\]{}]/g, '\\$&');

    const regex = new RegExp(`(${escapedSearchTerm})`, 'gi');
    return text.replace(regex, (match) => {
